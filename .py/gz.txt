import os
import json
import shutil
import re

def normalize_id(text):
    """Converts '2006/15591' to '2006_15591' for reliable matching."""
    match = re.search(r"(\d{4})/(\d+)", str(text))
    if match:
        return f"{match.group(1)}_{match.group(2)}"
    return str(text).strip()

def load_json(json_path):
    try:
        with open(json_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
            return [data] if isinstance(data, dict) else data
    except Exception as e:
        print(f"Error reading {json_path}: {e}")
        return []

def save_json(json_path, data):
    try:
        with open(json_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=4)
    except Exception as e:
        print(f"Error saving {json_path}: {e}")

def get_app_id(record):
    """Extracts and normalizes the application ID from a record."""
    if "APPLICATIONNO" in record and record["APPLICATIONNO"]:
        return normalize_id(record["APPLICATIONNO"])
    return None

def is_valid_data(value):
    """Checks if a value is actually meaningful (not empty, None, or 'null')."""
    if value is None:
        return False
    if value == "" or value == []:
        return False
    if isinstance(value, str) and value.strip().lower() == "null":
        return False
    if isinstance(value, dict) and not value:
        return False
    return True

def deep_merge(target, source):
    """Recursively merges source into target. Only overwrites if source data is valid."""
    merged = False
    for key, source_val in source.items():
        if isinstance(source_val, dict) and isinstance(target.get(key), dict):
            if deep_merge(target[key], source_val):
                merged = True
        elif is_valid_data(source_val):
            if target.get(key) != source_val:
                target[key] = source_val
                merged = True
    return merged

def process_direct_mapping(root_dir):
    OFFSET = 434  # Maps GZ_1 -> GZ_435
    
    global_stats = {
        "metadata_updated": 0, 
        "records_appended": 0,
        "images_copied": 0, 
        "images_skipped": 0
    }

    print(f"Starting deep merge: GZ_1 -> GZ_435 up to GZ_63 -> GZ_499...\n")
    print("-" * 60)

    for i in range(1, 64):
        src_folder_name = f"GZ_{i}"               
        dst_folder_name = f"GZ_{i + OFFSET}"      
        
        src_folder = os.path.join(root_dir, src_folder_name)
        dst_folder = os.path.join(root_dir, dst_folder_name)
        
        if not os.path.exists(src_folder):
            continue
            
        print(f"Processing {src_folder_name} -> {dst_folder_name} ...")
        
        # Local stats for this specific folder pair
        local_updated = 0
        local_appended = 0
        local_img_copied = 0
        local_img_skipped = 0
        
        # --- 1. MERGE METADATA ---
        src_json_path = os.path.join(src_folder, "metadata.json")
        dst_json_path = os.path.join(dst_folder, "metadata.json")
        
        if os.path.exists(src_json_path):
            src_records = load_json(src_json_path)
            dst_records = load_json(dst_json_path) if os.path.exists(dst_json_path) else []
            
            dst_map = {get_app_id(rec): rec for rec in dst_records if get_app_id(rec)}
            
            metadata_changed = False
            
            for src_rec in src_records:
                app_id = get_app_id(src_rec)
                if not app_id:
                    continue
                    
                if app_id in dst_map:
                    # Match found: overwrite valid data
                    if deep_merge(dst_map[app_id], src_rec):
                        local_updated += 1
                        global_stats["metadata_updated"] += 1
                        metadata_changed = True
                else:
                    # No match found: simply copy/append the whole record
                    dst_records.append(src_rec)
                    dst_map[app_id] = src_rec
                    local_appended += 1
                    global_stats["records_appended"] += 1
                    metadata_changed = True
            
            if metadata_changed:
                os.makedirs(dst_folder, exist_ok=True)
                save_json(dst_json_path, dst_records)

        # --- 2. COPY IMAGES ---
        src_images_dir = os.path.join(src_folder, "images")
        dst_images_dir = os.path.join(dst_folder, "images")
        
        if os.path.exists(src_images_dir):
            os.makedirs(dst_images_dir, exist_ok=True)
            
            for img_filename in os.listdir(src_images_dir):
                if not img_filename.lower().endswith(('.png', '.jpg', '.jpeg')):
                    continue
                    
                src_img_path = os.path.join(src_images_dir, img_filename)
                dst_img_path = os.path.join(dst_images_dir, img_filename)
                
                if not os.path.exists(dst_img_path):
                    try:
                        shutil.copy2(src_img_path, dst_img_path)
                        local_img_copied += 1
                        global_stats["images_copied"] += 1
                    except Exception as e:
                        print(f"  [!] Error copying {img_filename}: {e}")
                else:
                    local_img_skipped += 1
                    global_stats["images_skipped"] += 1

        # Print the results for this specific folder
        print(f"  -> Metadata: {local_updated} existing overwritten | {local_appended} new simply copied")
        print(f"  -> Images:   {local_img_copied} newly copied | {local_img_skipped} skipped (already existed)")
        print("-" * 60)

    # Print the final grand total at the very end
    print("\n" + "="*50)
    print("FINAL GRAND TOTALS")
    print("METADATA:")
    print(f"  Total existing records overwritten: {global_stats['metadata_updated']}")
    print(f"  Total new records simply copied:    {global_stats['records_appended']}")
    print("-" * 20)
    print("IMAGES:")
    print(f"  Total successfully copied:          {global_stats['images_copied']}")
    print(f"  Total skipped (already existed):    {global_stats['images_skipped']}")
    print("="*50)
